🚀 Linux Boot Process

BIOS/UEFI → GRUB2 → Kernel → systemd
StageComponentAction
1BIOS / UEFIPOST (Power-On Self Test), finds bootable device via MBR/GPT
2GRUB2Bootloader loads kernel image (vmlinuz) + initramfs into RAM
3KernelInitializes hardware, mounts initramfs as temporary root (/)
4initramfsLoads drivers, finds real root filesystem, hands off to systemd
5systemd (PID 1)Reads target units, starts services in parallel, reaches default.target

Retry Pattern

retry() {
  local max_attempts=3
  local delay=5
  local attempt=1
  local cmd=("$@")

  while (( attempt <= max_attempts )); do
    if "${cmd[@]}"; then
      return 0
    fi
    log "Attempt $attempt/$max_attempts failed. Retrying in ${delay}s..."
    sleep "$delay"
    (( attempt++ ))
    (( delay *= 2 ))  # exponential backoff
  done

  log "ERROR: Command failed after $max_attempts attempts: ${cmd[*]}"
  return 1
}

# Usage
retry curl -sf https://api.example.com/health

Lock File Pattern

LOCK_FILE="/var/run/deploy.lock"

acquire_lock() {
  exec 200>"$LOCK_FILE"
  if ! flock -n 200; then
    echo "Another instance is running. PID: $(cat $LOCK_FILE)"
    exit 1
  fi
  echo $$ > "$LOCK_FILE"
  log "Lock acquired (PID: $$)"
}

release_lock() {
  flock -u 200
  rm -f "$LOCK_FILE"
}

trap release_lock EXIT
acquire_lock

Complete Deployment Script

#!/usr/bin/env bash
set -euo pipefail
IFS=$'\n\t'

readonly APP_DIR="/var/www/myapp"
readonly BACKUP_DIR="/var/backups/myapp"
readonly SERVICE="myapp"
readonly REPO="git@github.com:company/myapp.git"

log() { echo "[$(date '+%Y-%m-%dT%H:%M:%S')] $*" | tee -a /var/log/deploy.log; }

backup_current() {
  local ts; ts=$(date +%Y%m%d_%H%M%S)
  if [[ -d "$APP_DIR" ]]; then
    tar -czf "$BACKUP_DIR/app_${ts}.tar.gz" "$APP_DIR"
    log "Backup created: app_${ts}.tar.gz"
  fi
}

deploy_code() {
  if [[ -d "$APP_DIR/.git" ]]; then
    git -C "$APP_DIR" pull origin main
  else
    git clone "$REPO" "$APP_DIR"
  fi
  log "Code deployed from $REPO"
}

install_deps() {
  cd "$APP_DIR"
  npm ci --only=production
  log "Dependencies installed"
}

restart_service() {
  systemctl restart "$SERVICE"
  sleep 3
  if systemctl is-active --quiet "$SERVICE"; then
    log "Service $SERVICE started successfully"
  else
    log "ERROR: Service failed to start"
    return 1
  fi
}

main() {
  log "=== Deployment started ==="
  backup_current
  deploy_code
  install_deps
  restart_service
  log "=== Deployment complete ==="
}

main "$@"

🔐 File Permissions, ACL & umask

Permissions Deep Dive
# Permission bits: rwxrwxrwx = owner/group/others
# r=4, w=2, x=1
chmod 755 script.sh        # rwxr-xr-x (owner full, group/others read+exec)
chmod 644 config.conf      # rw-r--r-- (owner read+write, others read)
chmod 600 ~/.ssh/id_rsa    # rw------- (private key, owner only)
chmod 777 /tmp/shared      # full access to all (dangerous)

# Symbolic mode
chmod u+x,g-w,o-r file
chmod a+r file             # a = all (u+g+o)
chmod -R 755 /var/www/     # recursive

# Special bits
chmod u+s /usr/bin/passwd  # SUID — runs as owner, not caller
chmod g+s /var/shared/     # SGID — new files inherit directory group
chmod +t /tmp/              # Sticky — only owner can delete their files

# Find SUID/SGID files (security audit)
find / -perm -4000 -type f 2>/dev/null   # SUID
find / -perm -2000 -type f 2>/dev/null   # SGID

# umask — default permission mask
umask                      # show current (usually 022)
umask 027                  # set: files=640, dirs=750
# Formula: file=666-umask, dir=777-umask

💾 Storage, LVM & RAID

LVM — Complete Workflow

LVM allows flexible disk management — resize volumes, take snapshots without downtime.

# LVM hierarchy: Physical Volumes (PV) → Volume Groups (VG) → Logical Volumes (LV)

# ── SETUP ──
# 1. Create Physical Volume
pvcreate /dev/sdb /dev/sdc
pvdisplay                            # show PV details
pvs                                  # summary

# 2. Create Volume Group
vgcreate data_vg /dev/sdb /dev/sdc
vgdisplay data_vg
vgs

# 3. Create Logical Volume
lvcreate -L 80G -n data_lv data_vg        # fixed size
lvcreate -l 100%FREE -n data_lv data_vg   # use all space
lvcreate -l 80%VG -n data_lv data_vg      # 80% of VG
lvdisplay

# 4. Create filesystem and mount
mkfs.ext4 /dev/data_vg/data_lv
mkfs.xfs /dev/data_vg/data_lv       # for XFS
mkdir /data
mount /dev/data_vg/data_lv /data

# Persistent mount in /etc/fstab
echo '/dev/data_vg/data_lv /data ext4 defaults 0 2' >> /etc/fstab
mount -a                             # test fstab

# ── EXTEND ──
# Add new disk to VG
pvcreate /dev/sdd
vgextend data_vg /dev/sdd

# Extend LV and resize filesystem
lvextend -L +20G /dev/data_vg/data_lv    # add 20GB
lvextend -l +100%FREE /dev/data_vg/data_lv  # use all available
resize2fs /dev/data_vg/data_lv           # ext4 online resize
xfs_growfs /data                          # XFS online resize

# ── SNAPSHOT ──
lvcreate -L 10G -s -n data_snap /dev/data_vg/data_lv  # snapshot
mount -o ro /dev/data_vg/data_snap /mnt/snap            # mount snapshot
# Revert to snapshot (data_lv must be unmounted)
lvconvert --merge /dev/data_vg/data_snap

# ── MIGRATE ──
# Move data from one PV to another
pvmove /dev/sdb /dev/sdd
# Then remove old PV
vgreduce data_vg /dev/sdb
pvremove /dev/sdb

Location Block Priority

# Priority order (= > ^~ > regex > prefix)
server {
  # 1. Exact match (highest priority)
  location = /health {
    return 200 "OK";
    access_log off;
  }

  # 2. Preferential prefix (stops regex search if matched)
  location ^~ /static/ {
    root /var/www;
    expires 30d;
    add_header Cache-Control "public, no-transform";
  }

  # 3. Case-sensitive regex
  location ~ \.php$ {
    fastcgi_pass unix:/run/php/php8.1-fpm.sock;
    include fastcgi_params;
    fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
  }

  # 4. Case-insensitive regex
  location ~* \.(jpg|jpeg|png|gif|webp|svg)$ {
    expires 1y;
    add_header Cache-Control "public, immutable";
    try_files $uri =404;
  }

  # 5. Prefix (lowest priority)
  location / {
    try_files $uri $uri/ /index.html;
  }
}

Upstream Load Balancing

# /etc/nginx/sites-available/myapp.conf
upstream app_servers {
  least_conn;                    # least-connections algorithm
  # or: ip_hash; (sticky sessions)
  # or: round_robin; (default)

  server 10.0.1.10:3000 weight=3 max_fails=3 fail_timeout=30s;
  server 10.0.1.11:3000 weight=1;
  server 10.0.1.12:3000 backup;  # only used if primary fails
}

server {
  listen 443 ssl http2;
  server_name myapp.com;

  ssl_certificate     /etc/ssl/myapp.crt;
  ssl_certificate_key /etc/ssl/myapp.key;
  ssl_protocols TLSv1.2 TLSv1.3;
  ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
  ssl_prefer_server_ciphers off;
  ssl_session_cache shared:SSL:10m;
  ssl_session_timeout 1d;

  # Security headers
  add_header Strict-Transport-Security "max-age=63072000" always;
  add_header X-Content-Type-Options nosniff;
  add_header X-Frame-Options DENY;
  add_header Content-Security-Policy "default-src 'self'";

  location / {
    proxy_pass http://app_servers;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_connect_timeout 5s;
    proxy_read_timeout 30s;
    proxy_buffer_size 4k;
    proxy_buffers 8 8k;
  }
}

🎯 Linux Interview Questions

Top 20 Linux Admin Interview Q&A

Q1: What happens when you type a command in Linux?

Shell reads input → checks aliases → checks shell builtins → searches PATH directories in order → forks a child process (fork()) → child calls execve() to replace itself with the command binary → parent shell waits (waitpid()).

Q2: What is the difference between hard link and symbolic link?

Hard link: Second directory entry pointing to same inode. Cannot span filesystems. Still works if original deleted. ls -li shows same inode number.
Soft/Symbolic link: Points to the file path (not inode). Dangling link if original deleted. Can span filesystems and link to directories.

Q3: How do you check which process is listening on port 80?

ss -tlnp | grep :80
lsof -i :80
fuser 80/tcp

Q4: Explain the difference between kill -9 and kill -15

kill -15 (SIGTERM): Polite shutdown. Process can catch it, clean up, close files, flush buffers before exiting.
kill -9 (SIGKILL): Cannot be caught or ignored. Kernel immediately terminates process. Use as last resort — may leave resources in bad state.

Q5: What is zombie process? How to handle it?

Zombie = process that finished but parent hasn't read its exit status (wait()). Shows as Z in ps. Fix: kill parent (parent will reap zombies) or if parent is init it auto-reaps. ps aux | grep Z

Q6: How does LVM snapshot work?

LVM snapshot uses Copy-on-Write (CoW). When a block in the original LV is about to be modified, LVM first copies the original block to the snapshot LV, then allows the write. Snapshot contains only changed blocks = efficient.

Q7: What is the difference between ext4 and XFS?

Featureext4XFS
Max filesystem1 EiB8 EiB
Max file size16 TiB8 EiB
Shrink onlineYesNo
Large filesGoodExcellent
Default onDebian/UbuntuRHEL 7+

Q8: Explain SELinux context format

user:role:type:level. Type is most important for policy decisions (Type Enforcement). Files get context from policy rules. httpd process runs in httpd_t domain and can only access files with matching types like httpd_sys_content_t.

Q9: How would you find all SUID files on the system?

find / -perm -4000 -type f 2>/dev/null
find / -perm /6000 -type f 2>/dev/null  # SUID or SGID

Q10: What does the sticky bit do on a directory?

Sticky bit on a directory (like /tmp) means only the file owner, directory owner, or root can delete/rename files in that directory — even if others have write permission. Prevents users from deleting each other's files.

🔬 eBPF — Linux Superpower

Observability, Security & Networking without Kernel Modules

What is eBPF?

  • eBPF (extended Berkeley Packet Filter) — run sandboxed programs in the Linux kernel without changing kernel source or loading modules
  • Use cases: observability (Cilium Hubble), security (Falco, Tetragon), networking (Cilium CNI), performance profiling (bpftrace, Parca)
  • Programs attach to kernel hooks: kprobes, uprobes, tracepoints, XDP (eXpress Data Path), tc (traffic control), LSM (Linux Security Module)

bpftrace — One-liners for System Insight

# List all available probes
bpftrace -l 'tracepoint:syscalls:*' | head -20

# Trace all open() syscalls with filename
bpftrace -e 'tracepoint:syscalls:sys_enter_openat { printf("%s opened %s\n", comm, str(args->filename)); }'

# Histogram of read() sizes by process
bpftrace -e 'tracepoint:syscalls:sys_exit_read { @[comm] = hist(args->ret); }'

# Top 10 slowest disk I/Os (latency > 1ms)
bpftrace -e '
kprobe:blk_account_io_start { @start[arg0] = nsecs; }
kprobe:blk_account_io_done / @start[arg0] / {
  $lat = (nsecs - @start[arg0]) / 1000;
  if ($lat > 1000) { printf("I/O latency: %d us\n", $lat); }
  delete(@start[arg0]);
}'

# TCP connection latency
bpftrace -e 'kprobe:tcp_rcv_state_process /args->sk->__sk_common.skc_state == 2/ { @[comm] = hist((nsecs - @start[args->sk]) / 1000); }'

# Network packet drops
bpftrace -e 'kprobe:kfree_skb { @drops[kstack(5)] = count(); } interval:s:5 { print(@drops); clear(@drops); }'
