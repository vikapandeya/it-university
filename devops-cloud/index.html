Bash Scripting for DevOps

Production bash scripts must be robust, observable, and safe.

#!/usr/bin/env bash
# deploy.sh — Production deployment script
set -euo pipefail          # exit on error, unbound var, pipe failure
IFS=$'\n\t'                # safer word splitting

# ── LOGGING ──────────────────────────────────────────
readonly LOG_FILE="/var/log/deploy.log"
readonly TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

log()   { echo "[${TIMESTAMP}] [INFO]  $*" | tee -a "$LOG_FILE"; }
warn()  { echo "[${TIMESTAMP}] [WARN]  $*" | tee -a "$LOG_FILE" >&2; }
error() { echo "[${TIMESTAMP}] [ERROR] $*" | tee -a "$LOG_FILE" >&2; exit 1; }

# ── ARGUMENT PARSING ─────────────────────────────────
usage() {
  cat <<EOF
Usage: $0 [OPTIONS]
Options:
  -e, --env       Environment (dev|staging|prod)   [required]
  -v, --version   Docker image version tag         [required]
  -n, --dry-run   Print commands without executing
  -h, --help      Show this help
EOF
}

ENV=""
VERSION=""
DRY_RUN=false

while [[ $# -gt 0 ]]; do
  case $1 in
    -e|--env)     ENV="$2";     shift 2 ;;
    -v|--version) VERSION="$2"; shift 2 ;;
    -n|--dry-run) DRY_RUN=true; shift   ;;
    -h|--help)    usage; exit 0          ;;
    *)            error "Unknown option: $1" ;;
  esac
done

[[ -z "$ENV" ]]     && error "--env is required"
[[ -z "$VERSION" ]] && error "--version is required"
[[ "$ENV" =~ ^(dev|staging|prod)$ ]] || error "Invalid env: $ENV"

# ── SIGNAL HANDLING (cleanup on exit/ctrl+c) ─────────
TMPDIR_WORK=$(mktemp -d)
cleanup() {
  log "Cleaning up temp dir: $TMPDIR_WORK"
  rm -rf "$TMPDIR_WORK"
}
trap cleanup EXIT
trap 'error "Script interrupted"' INT TERM

# ── RETRY FUNCTION ────────────────────────────────────
retry() {
  local -r max_attempts=${RETRY_MAX:-3}
  local -r delay=${RETRY_DELAY:-5}
  local attempt=1
  until "$@"; do
    if (( attempt == max_attempts )); then
      error "Command failed after $max_attempts attempts: $*"
    fi
    warn "Attempt $attempt failed. Retrying in ${delay}s..."
    sleep "$delay"
    (( attempt++ ))
  done
}

# ── RUN (respects --dry-run) ─────────────────────────
run() {
  log "Running: $*"
  if [[ "$DRY_RUN" == true ]]; then
    log "[DRY-RUN] Skipping execution"
    return 0
  fi
  "$@"
}

# ── MAIN DEPLOYMENT LOGIC ────────────────────────────
NAMESPACE="$ENV"
DEPLOYMENT="myapp"
IMAGE="123456789.dkr.ecr.us-east-1.amazonaws.com/myapp:${VERSION}"

log "Starting deployment: env=$ENV version=$VERSION"

# Verify image exists in ECR
run aws ecr describe-images \
  --repository-name myapp \
  --image-ids imageTag="$VERSION" \
  --region us-east-1 >/dev/null

# Update kubeconfig
run aws eks update-kubeconfig \
  --name "cluster-$ENV" \
  --region us-east-1

# Apply manifests from temp dir
cp -r ./k8s/"$ENV"/* "$TMPDIR_WORK/"
sed -i "s|IMAGE_PLACEHOLDER|${IMAGE}|g" "$TMPDIR_WORK"/*.yaml

run kubectl apply -f "$TMPDIR_WORK/" -n "$NAMESPACE"

# Wait for rollout with retry
retry kubectl rollout status deployment/"$DEPLOYMENT" \
  -n "$NAMESPACE" --timeout=5m

log "Deployment complete: $ENV @ $VERSION"

Merge vs Rebase vs Squash
# ── MERGE (preserves history, adds merge commit) ────
git checkout main
git merge feature/my-feature
# Use when: preserving full context, release branches

# ── REBASE (linear history, rewrites commits) ────────
git checkout feature/my-feature
git rebase main         # replays feature commits on top of main
git checkout main
git merge feature/my-feature  # fast-forward
# Use when: keeping main history clean, local cleanup
# NEVER rebase shared/public branches

# ── SQUASH (one commit per feature) ──────────────────
git merge --squash feature/my-feature
git commit -m "feat: add user authentication"
# Use when: noisy feature branch, clean PR history desired

# ── INTERACTIVE REBASE (edit history) ────────────────
git rebase -i HEAD~5   # edit last 5 commits
# Commands: pick, reword, edit, squash, fixup, drop

  echo "  type(scope): description"
  echo "  e.g.: feat(auth): add JWT refresh tokens"
  exit 1
fi

Multi-Stage Dockerfile
# ── Stage 1: Builder ────────────────────────────────
FROM node:20-alpine AS builder
WORKDIR /build

# Copy dependency files first (layer caching)
COPY package*.json ./
RUN npm ci                    # clean install, uses package-lock.json

# Copy source and build
COPY . .
RUN npm run build             # outputs to /build/dist

# ── Stage 2: Production ──────────────────────────────
FROM node:20-alpine AS production
WORKDIR /app

# Only copy production deps + built artifacts
COPY package*.json ./
RUN npm ci --only=production && npm cache clean --force

COPY --from=builder /build/dist ./dist
COPY --from=builder /build/public ./public

# Security: non-root user
RUN addgroup -S appgroup && adduser -S appuser -G appgroup
USER appuser

EXPOSE 3000
HEALTHCHECK --interval=30s --timeout=5s --retries=3 \
  CMD wget -qO- http://localhost:3000/health || exit 1

CMD ["node", "dist/server.js"]

# Result: ~80MB instead of ~900MB (no dev deps, no source)
Docker Networking
# ── BRIDGE (default) ────────────────────────────────
# Containers on same bridge can communicate by container name
# Use for: single-host multi-container apps
docker network create app-net
docker run -d --network app-net --name db postgres:15
docker run -d --network app-net --name web myapp
# web can reach db at hostname "db"

# ── HOST ─────────────────────────────────────────────
# Container shares host's network stack (no isolation)
# Use for: performance-critical apps, monitoring agents
docker run --network host prometheus

# ── OVERLAY ──────────────────────────────────────────
# Multi-host networking (Docker Swarm / encrypted tunnel)
# Use for: distributed apps across multiple Docker hosts
docker network create -d overlay --attachable my-overlay

# ── NONE ─────────────────────────────────────────────
# Completely isolated, no network interfaces
# Use for: batch jobs, security-sensitive tasks
docker run --network none my-batch-job

Full CI/CD Workflow with OIDC Auth
# .github/workflows/ci-cd.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

permissions:
  id-token: write    # Required for OIDC
  contents: read
  security-events: write

env:
  AWS_REGION: us-east-1
  ECR_REPOSITORY: myapp
  EKS_CLUSTER: production

jobs:
  # ── JOB 1: Test Matrix ──────────────────────────────
  test:
    name: Test (Node ${{ matrix.node-version }})
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [18, 20]
      fail-fast: false
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run linter
        run: npm run lint

      - name: Run unit tests
        run: npm test -- --coverage

      - name: Upload coverage
        uses: actions/upload-artifact@v4
        with:
          name: coverage-node${{ matrix.node-version }}
          path: coverage/
          retention-days: 7

  # ── JOB 2: Security Scan ────────────────────────────
  security-scan:
    name: Security Scan (Trivy)
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Build image for scanning
        run: docker build -t scan-target:${{ github.sha }} .

      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: scan-target:${{ github.sha }}
          format: sarif
          output: trivy-results.sarif
          severity: HIGH,CRITICAL
          exit-code: 1

      - name: Upload Trivy results to GitHub Security
        uses: github/codeql-action/upload-sarif@v3
        with:
          sarif_file: trivy-results.sarif

  # ── JOB 3: Build & Push to ECR (OIDC — no static keys) ─
  build-and-push:
    name: Build and Push to ECR
    runs-on: ubuntu-latest
    needs: [test, security-scan]
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    environment: production
    outputs:
      image-tag: ${{ steps.meta.outputs.tags }}
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials (OIDC)
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: arn:aws:iam::123456789:role/github-actions-ecr
          aws-region: ${{ env.AWS_REGION }}
          # No AWS_ACCESS_KEY_ID or AWS_SECRET_ACCESS_KEY needed!

      - name: Login to Amazon ECR
        id: login-ecr
        uses: aws-actions/amazon-ecr-login@v2

      - name: Extract Docker metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ steps.login-ecr.outputs.registry }}/${{ env.ECR_REPOSITORY }}
          tags: |
            type=sha,prefix=,format=short
            type=semver,pattern={{version}}
            type=raw,value=latest,enable={{is_default_branch}}

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
          build-args: |
            BUILD_DATE=${{ github.event.head_commit.timestamp }}
            GIT_SHA=${{ github.sha }}

  # ── JOB 4: Deploy to Kubernetes ─────────────────────
  deploy:
    name: Deploy to Kubernetes
    runs-on: ubuntu-latest
    needs: build-and-push
    environment:
      name: production
      url: https://myapp.example.com
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials (OIDC)
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: arn:aws:iam::123456789:role/github-actions-eks
          aws-region: ${{ env.AWS_REGION }}

      - name: Update kubeconfig
        run: aws eks update-kubeconfig --name ${{ env.EKS_CLUSTER }} --region ${{ env.AWS_REGION }}

      - name: Deploy with kubectl
        run: |
          IMAGE="${{ needs.build-and-push.outputs.image-tag }}"
          kubectl set image deployment/myapp \
            myapp="$IMAGE" \
            -n production
          kubectl rollout status deployment/myapp -n production --timeout=5m

      - name: Smoke test
        run: |
          sleep 10
          curl -f https://myapp.example.com/health || exit 1

              -Dsonar.projectKey=${APP_NAME} \
              -Dsonar.sources=src \
              -Dsonar.javascript.lcov.reportPaths=coverage/lcov.info
          """
        }
        timeout(time: 5, unit: 'MINUTES') {
          waitForQualityGate abortPipeline: true
        }
      }
    }

    stage('Docker Build') {
      steps {
        sh "docker build -t ${IMAGE_TAG} ."
        sh "trivy image --exit-code 1 --severity CRITICAL ${IMAGE_TAG}"
      }
    }

    stage('ECR Push') {
      when {
        branch 'main'
      }
      steps {
        sh """
          aws ecr get-login-password --region us-east-1 \
            | docker login --username AWS --password-stdin ${ECR_REGISTRY}
          docker push ${IMAGE_TAG}
          docker tag ${IMAGE_TAG} ${ECR_REGISTRY}/${APP_NAME}:latest
          docker push ${ECR_REGISTRY}/${APP_NAME}:latest
        """
      }
    }

    stage('Deploy to Kubernetes') {
      when {
        branch 'main'
      }
      steps {
        withKubeConfig([credentialsId: 'eks-kubeconfig']) {
          sh """
            kubectl set image deployment/${APP_NAME} \
              ${APP_NAME}=${IMAGE_TAG} \
              -n production
            kubectl rollout status deployment/${APP_NAME} \
              -n production --timeout=5m
          """
        }
      }
    }

    stage('Smoke Test') {
      when {
        branch 'main'
      }
      steps {
        sh 'sleep 15'
        sh 'curl -f https://myapp.example.com/health'
      }
    }

    stage('Notify') {
      steps {
        script {
          def color = currentBuild.result == 'SUCCESS' ? 'good' : 'danger'
          slackSend(
            color: color,
            message: "${APP_NAME} #${BUILD_NUMBER} ${currentBuild.result}: ${env.BUILD_URL}"
          )
        }
      }
    }
  }

  post {
    always {
      cleanWs()
      sh "docker rmi ${IMAGE_TAG} || true"
    }
    success {
      echo "Pipeline succeeded for build #${env.BUILD_NUMBER}"
    }
    failure {
      emailext(
        subject: "FAILURE: ${APP_NAME} Build #${env.BUILD_NUMBER}",
        body: "Check console: ${env.BUILD_URL}",
        to: 'devops-team@example.com'
      )
    }
  }

Observability

Three Pillars: Metrics, Logs, Traces

Observability lets you understand system behavior from its outputs without needing to re-deploy or add ad-hoc instrumentation.

  • Metrics — numeric time-series data. Aggregatable. Cheap to store. Best for dashboards and alerting. Tools: Prometheus, Grafana, Datadog.
  • Logs — structured/unstructured events. Rich context. Expensive at scale. Best for debugging specific issues. Tools: Loki, ELK, CloudWatch Logs.
  • Traces — request flow across services. Latency breakdown. Best for finding bottlenecks in distributed systems. Tools: Jaeger, Tempo, X-Ray.

SRE Core Concepts

ConceptDefinitionExample
SLI (Service Level Indicator)A quantitative measure of service behavior99th percentile request latency

Error Budget Policy

  • If error budget healthy (>50%) — fast releases, experiments allowed
  • If error budget at risk (10–50%) — slow releases, focus on reliability
  • If error budget exhausted (<0%) — freeze releases, mandatory reliability sprint
  • This creates natural alignment between dev velocity and system reliability
